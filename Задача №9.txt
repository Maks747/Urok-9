import java.util.Scanner;


public class Main {
    public static void main(String[] args) {

        int a, b;

        Scanner in = new Scanner(System.in);
        System.out.print("Введите количество строк массива: ");
        a = in.nextInt();
        System.out.print("Введите количество столбцов массива: ");
        b = in.nextInt();

        double [][] arr;
        arr = new double [a][b];

        int kol = 0;
        kol = a * b;

        System.out.println("Количество элементов в массиве: " + kol);

        double resultat = 0;
        double resultat1 = 0;

        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[i].length; j++) {
                System.out.print("Введите элемент arr[" + i + "][" + j + "]: ");
                arr[i][j] = in.nextDouble();
                resultat += arr[i][j];
            }
        }
        resultat1 = resultat / kol;
        System.out.println("Значение среднего арифметического равно: " + resultat1);

        System.out.println("Умножение элемента массива на среднее арифметическое: ");
        for (int i = 0;i < arr.length; i++) {
            for (int j = 0; j < arr[i].length; j++) {
                System.out.println( arr[i][j] * resultat1 + " ");
            }
        }
    }
}
